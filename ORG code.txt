    <script>
        const encodeInput = document.getElementById('encodeInput');
        const decodeInput = document.getElementById('decodeInput');
        const statusMsg = document.getElementById('statusMsg');
        const mainApp = document.getElementById('mainApp');
        const pwdModal = document.getElementById('pwdModal');
        const pwdInput = document.getElementById('pwdInput');
        const pwdError = document.getElementById('pwdError');
        const viewerModal = document.getElementById('viewerModal');
        const viewerTitle = document.getElementById('viewerTitle');
        const viewerText = document.getElementById('viewerText');
        const viewerVideo = document.getElementById('viewerVideo');
        const viewerAudio = document.getElementById('viewerAudio');
        const viewerAudioContainer = document.getElementById('viewerAudioContainer');

        let pendingAction = null;
        const PROTOCOL_HASH = 78757006; 
        const DELIMITER = "|||SURAJ_PROTOCOL_V2|||";

        function generateHash(str) {
            let hash = 0;
            for (let i = 0; i < str.length; i++) {
                hash = (hash << 5) - hash + str.charCodeAt(i);
                hash |= 0;
            }
            return hash;
        }

        function requestAccess(action) {
            pendingAction = action;
            pwdInput.value = '';
            pwdError.classList.add('hidden');
            pwdModal.classList.remove('hidden');
            pwdInput.focus();
        }

        function cancelAccess() {
            pwdModal.classList.add('hidden');
            pendingAction = null;
        }

        function verifyAccess() {
            const inputHash = generateHash(pwdInput.value);
            if (inputHash === PROTOCOL_HASH) {
                pwdModal.classList.add('hidden');
                if (pendingAction === 'encode') encodeInput.click();
                else if (pendingAction === 'decode') decodeInput.click();
            } else {
                pwdError.classList.remove('hidden');
                pwdInput.value = '';
                pwdInput.focus();
            }
        }

        pwdInput.addEventListener('keypress', function (e) {
            if (e.key === 'Enter') verifyAccess();
        });

        function stringToHex(str) {
            const encoder = new TextEncoder();
            const view = encoder.encode(str); 
            let hex = '';
            for (let i = 0; i < view.length; i++) hex += view[i].toString(16).padStart(2, '0');
            return hex;
        }

        function hexToString(hex) {
            if (hex.length % 2 !== 0) throw new Error("Invalid");
            const bytes = new Uint8Array(hex.length / 2);
            for (let i = 0; i < hex.length; i += 2) bytes[i / 2] = parseInt(hex.substring(i, i + 2), 16);
            return new TextDecoder().decode(bytes);
        }

        function updateStatus(message, isError = false, isProcessing = false) {
            statusMsg.textContent = message;
            if (isProcessing) {
                statusMsg.className = "text-[11px] font-mono text-yellow-400 font-bold tracking-widest animate-pulse";
                mainApp.classList.add('processing-fx');
            } else if (isError) {
                statusMsg.className = "text-[11px] font-mono text-red-400 font-bold tracking-widest";
                mainApp.classList.remove('processing-fx');
            } else {
                statusMsg.className = "text-[11px] font-mono text-emerald-400 font-bold tracking-widest";
                mainApp.classList.remove('processing-fx');
            }
        }

        encodeInput.addEventListener('change', function(event) {
            const file = event.target.files[0];
            if (!file) return;
            updateStatus("ENCRYPTING...", false, true);
            const reader = new FileReader();
            reader.onload = function(e) {
                setTimeout(() => {
                    try {
                        const payloadStr = file.name + DELIMITER + e.target.result;
                        const hexData = stringToHex(payloadStr);
                        const baseName = file.name.replace(/\.[^/.]+$/, "");
                        const newFileName = baseName + ".suraj";
                        const blob = new Blob([hexData], { type: "text/plain" });
                        const url = URL.createObjectURL(blob);
                        const a = document.createElement('a');
                        a.href = url;
                        a.download = newFileName;
                        document.body.appendChild(a);
                        a.click();
                        document.body.removeChild(a);
                        URL.revokeObjectURL(url);
                        updateStatus(`ENCRYPTED: ${newFileName}`);
                    } catch (error) {
                        updateStatus(`ERROR`, true);
                    }
                    encodeInput.value = ''; 
                }, 100);
            };
            reader.readAsDataURL(file);
        });

        decodeInput.addEventListener('change', function(event) {
            const file = event.target.files[0];
            if (!file) return;
            updateStatus("DECRYPTING...", false, true);
            const reader = new FileReader();
            reader.onload = function(e) {
                setTimeout(() => {
                    try {
                        const hexData = e.target.result.trim(); 
                        const decodedPayload = hexToString(hexData);
                        const parts = decodedPayload.split(DELIMITER);
                        if (parts.length !== 2) throw new Error("Format");
                        const originalName = parts[0];
                        const dataUrl = parts[1];
                        const mimeMatch = dataUrl.match(/^data:(.*?);/);
                        const mimeType = mimeMatch ? mimeMatch[1].toLowerCase() : '';
                        const isText = mimeType.startsWith('text/') || originalName.toLowerCase().endsWith('.txt');
                        const isVideo = mimeType.startsWith('video/') || originalName.toLowerCase().endsWith('.mp4');
                        const isAudio = mimeType.startsWith('audio/') || originalName.toLowerCase().endsWith('.mp3');

                        if (isText) {
                            fetch(dataUrl).then(res => res.text()).then(text => { showInViewer('text', text, originalName); updateStatus(`VIEWING: ${originalName}`); });
                        } else if (isVideo) {
                            showInViewer('video', dataUrl, originalName);
                            updateStatus(`VIEWING: ${originalName}`);
                        } else if (isAudio) {
                            showInViewer('audio', dataUrl, originalName);
                            updateStatus(`LISTENING: ${originalName}`);
                        } else {
                            const a = document.createElement('a');
                            a.href = dataUrl;
                            a.download = originalName;
                            document.body.appendChild(a);
                            a.click();
                            document.body.removeChild(a);
                            updateStatus(`DOWNLOADED: ${originalName}`);
                        }
                    } catch (error) {
                        updateStatus(`DECRYPTION FAILED`, true);
                    }
                    decodeInput.value = '';
                }, 100);
            };
            reader.readAsText(file);
        });

        function showInViewer(type, content, title) {
            viewerText.classList.add('hidden');
            viewerVideo.classList.add('hidden');
            viewerAudioContainer.classList.add('hidden');
            viewerTitle.textContent = `DECRYPTED: ${title}`;
            if (type === 'text') {
                viewerText.value = content;
                viewerText.classList.remove('hidden');
            } else if (type === 'video') {
                viewerVideo.src = content;
                viewerVideo.classList.remove('hidden');
            } else if (type === 'audio') {
                viewerAudio.src = content;
                viewerAudioContainer.classList.remove('hidden');
            }
            viewerModal.classList.remove('hidden');
        }

        function closeViewer() {
            viewerModal.classList.add('hidden');
            viewerText.value = ''; 
            viewerVideo.pause();
            viewerVideo.removeAttribute('src');
            viewerVideo.load();
            viewerAudio.pause();
            viewerAudio.removeAttribute('src');
            viewerAudio.load();
        }

        window.addEventListener('click', function(event) {
            if (event.target === viewerModal) closeViewer();
            if (event.target === pwdModal) cancelAccess();
        });
    </script>
